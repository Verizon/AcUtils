331  public async Task<bool> initAsync(AcDepot depot, AcStream stream = null, bool includeHidden = false)
332  {
333  bool ret = false; // assume failure
334  try
335  {
336  string cmd = null;
337  if (stream != null && includeHidden)
338  cmd = String.Format(@"getproperty -fix -s ""{0}""", stream);
339  else if (stream != null && !includeHidden)
340  cmd = String.Format(@"getproperty -fx -s ""{0}""", stream);
341  else if (includeHidden) // request is for all streams in depot including those that are hidden
342  cmd = String.Format(@"getproperty -fix -ks -p ""{0}""", depot);
343  else // request is for all streams except those that are hidden
344  cmd = String.Format(@"getproperty -fx -ks -p ""{0}""", depot);
345 
346  AcResult r = await AcCommand.runAsync(cmd).ConfigureAwait(false);
347  if (r != null && r.RetVal == 0)
348  {
349  XElement xml = XElement.Parse(r.CmdResult);
350  IEnumerable<XElement> query = from element in xml.Descendants("property") select element;
351  foreach (XElement e in query)
352  {
353  AcProperty property = new AcProperty();
354  string kind = (string)e.Attribute("kind");
355  property.Kind = (PropKind)Enum.Parse(typeof(PropKind), kind);
356  property.Depot = depot;
357  property.ID = (int)e.Attribute("streamNumber");
358  property.Name = (string)e.Attribute("streamName");
359  property.PropName = (string)e.Attribute("propertyName");
360  property.PropValue = (string)e;
361  lock (_locker) { Add(property); }
362  }
363 
364  ret = true; // operation succeeded
365  }
366  }
