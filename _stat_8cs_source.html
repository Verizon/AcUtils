327  public string ToString(string format, IFormatProvider provider)
352  {
353  if (provider != null)
354  {
355  ICustomFormatter fmt = provider.GetFormat(this.GetType()) as ICustomFormatter;
356  if (fmt != null)
357  return fmt.Format(format, this, provider);
358  }
359 
360  if (String.IsNullOrEmpty(format))
361  format = "G";
362 
363  switch (format.ToUpperInvariant())
364  {
365  case "LV": // long version (verbose)
366  {
367  string text;
368  if (_modTime != null)
369  {
370  text = String.Format("{1}, {2}{0}\tEID: {3} {{{4}}}, Size: {5}, ModTime: {6},{0}\t{7}, Virtual: {8}\\{9}, Real: {10}\\{11}",
371  Environment.NewLine, Location, Status, EID, ElementType, Size, ModTime, NamedVersion,
372  VirStreamNumber, VirVersionNumber, RealStreamNumber, RealVersionNumber);
373  }
374  else
375  {
376  text = String.Format("{1}, {2}{0}\tEID: {3} {{{4}}}{0}\t{5}, Virtual: {6}\\{7}, Real: {8}\\{9}",
377  Environment.NewLine, Location, Status, EID, ElementType, NamedVersion,
378  VirStreamNumber, VirVersionNumber, RealStreamNumber, RealVersionNumber);
379  }
380 
381  return text;
382  }
383  case "G": // location, the depot-relative path of the element (default when not using a format specifier)
384  return Location;
385  case "F": // True if the element is a folder, False otherwise
386  return Folder.ToString();
387  case "E": // UNIX/Linux systems only: True if the executable bit is set, False if cleared
388  return Executable.ToString();
389  case "I": // element ID
390  return EID.ToString();
391  case "T": // element's type: dir, text, binary, ptext, elink, or slink
392  return ElementType.ToString();
393  case "FS": // file size in bytes
394  return Size.ToString();
395  case "MT": // element's modification time
396  return ModTime.ToString();
397  case "H": // hierarchy type: parallel or serial
398  return HierType;
399  case "V": // virtual stream\\version number designation, e.g. 5\12
400  return String.Format("{0}\\{1}", VirStreamNumber, VirVersionNumber);
401  case "R": // real stream\\version number designation, e.g. 5\12
402  return String.Format("{0}\\{1}", RealStreamNumber, RealVersionNumber);
403  case "N": // named stream\version number format, e.g. MARS_STAGE\7 or MARS_STAGE_barnyrd\24
404  return NamedVersion;
405  case "L": // stream where element is located when it has (\e underlap)(\e member) or (\e overlap)(\e member) status
406  return LapStream;
407  case "TB": // first time-based stream found when 'stat -s stream -o -B -fox' is used to retrieve elements with (overlap) and/or (underlap) status
408  return TimeBasedStream;
409  case "S": // version's status, e.g. (kept)(member)
410  return Status;
411  default:
412  throw new FormatException(String.Format("The {0} format string is not supported.", format));
413  }

487  public static bool init(string xml)
488  {
489  bool ret = false; // assume failure
490  try
491  {
492  XElement elements = XElement.Parse(xml);
493  IEnumerable<XElement> query = from e in elements.Descendants("element")
494  where !e.Attribute("status").Value.Contains("no such elem")
495  select e;
496  foreach (XElement e in query)
497  {
498  Element element = new Element();
499  element.Status = (string)e.Attribute("status") ?? String.Empty;
500  element.Location = (string)e.Attribute("location") ?? String.Empty;
501  string dir = (string)e.Attribute("dir") ?? String.Empty;
502  element.Folder = String.Equals(dir, "yes");
503  string exe = (string)e.Attribute("executable") ?? String.Empty;
504  element.Executable = String.Equals(exe, "yes");
505  element.EID = (int?)e.Attribute("id") ?? 0;
506  string etype = (string)e.Attribute("elemType") ?? String.Empty;
507  element.ElementType = String.IsNullOrEmpty(etype) ?
508  ElementType.unknown : (ElementType)Enum.Parse(typeof(ElementType), etype);
509  element.Size = (long?)e.Attribute("size") ?? 0;
510  long modtime = (long?)e.Attribute("modTime") ?? 0;
511  if (modtime != 0)
512  element.ModTime = AcDateTime.AcDate2DateTime(modtime);
513  element.HierType = (string)e.Attribute("hierType") ?? String.Empty;
514  int ival;
515  string vir = (string)e.Attribute("Virtual") ?? String.Empty;
516  if (!String.IsNullOrEmpty(vir))
517  {
518  string[] arrVir = vir.Split('\\');
519  if (Int32.TryParse(arrVir[0], NumberStyles.Integer, null, out ival))
520  element.VirStreamNumber = ival;
521  if (Int32.TryParse(arrVir[1], NumberStyles.Integer, null, out ival))
522  element.VirVersionNumber = ival;
523  }
524  element.NamedVersion = (string)e.Attribute("namedVersion") ?? String.Empty;
525  string real = (string)e.Attribute("Real") ?? String.Empty;
526  if (!String.IsNullOrEmpty(real))
527  {
528  string[] arrReal = real.Split('\\');
529  if (Int32.TryParse(arrReal[0], NumberStyles.Integer, null, out ival))
530  element.RealStreamNumber = ival;
531  if (Int32.TryParse(arrReal[1], NumberStyles.Integer, null, out ival))
532  element.RealVersionNumber = ival;
533  }
534  element.LapStream = (string)e.Attribute("overlapStream") ?? String.Empty;
535  element.TimeBasedStream = (string)e.Attribute("timeBasisStream") ?? String.Empty;
536  lock (_locker) { _elements.Add(element); }
537  }
538 
539  ret = true; // operation succeeded
540  }
541 
542  catch (Exception ecx)
543  {
544  string msg = String.Format("Exception caught and logged in Stat.init{0}{1}",
545  Environment.NewLine, ecx.Message);
546  AcDebug.Log(msg);
547  }
548 
549  return ret;
550  }
