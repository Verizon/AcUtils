506  private bool initList(IEnumerable<XElement> query)
507  {

509  try

512  {
513  AcLock lk = new AcLock();
514  // Kind and Name are the only ones that always exist in the XML
515  // others exist only if there are values for them
516  string kind = (string)e.Attribute("kind");
517  lk.Kind = (LockKind)Enum.Parse(typeof(LockKind), kind);
518  lk.Name = (string)e.Attribute("Name");
519  string type = (string)e.Attribute("userType") ?? String.Empty;
520  lk.Type = String.IsNullOrEmpty(type) ? PrinType.none :
521  (PrinType)Enum.Parse(typeof(PrinType), type);
522  lk.ExceptFor = (string)e.Attribute("exceptFor") ?? String.Empty;
523  lk.OnlyFor = (string)e.Attribute("onlyFor") ?? String.Empty;
524  lk.Comment = (string)e.Attribute("comment") ?? String.Empty;
525  lock (_locker) { Add(lk); }
