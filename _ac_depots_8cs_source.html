767  public async Task<string> canViewAsync(AcUser user)
768  {
769  // create permissions object as a singleton (thread safe)
770  // see "C# Lazy Initialization && Race-to-initialize" http://stackoverflow.com/questions/11555755/c-sharp-lazy-initialization-race-to-initialize
771  await LazyInitializer.EnsureInitialized(ref _depotPermObj, ref _permInit, ref _permSync,
772  async () =>
773  {
774  _permissions = new AcPermissions(PermKind.depot);
775  return await _permissions.initAsync().ConfigureAwait(false);
776  }).ConfigureAwait(false);
777 
778  if (_depotPermObj.Result == false)
779  return null; // initialization failure, check log file
780 
781  // get the membership list for the user (includes explicit and implicit membership)
782  SortedSet<string> members = user.Principal.Members;
783  // list that will contain the depots this user has permission to view
784  List<string> canView = new List<string>();
785 
786  foreach (AcDepot depot in AsReadOnly())
787  {
788  bool isInheritable = false;
789  // Any() returns true if the collection contains one or more items that satisfy the condition defined by the predicate
790  // first, is an 'all' or 'none' permission explicitly set for this user on this depot?
791  if (_permissions.Any(
792  delegate(AcPermission p)
793  {
794  bool all = (p.Type == PermType.user && p.Rights == PermRights.all &&
795  String.Equals(p.AppliesTo, user.Principal.Name) && String.Equals(p.Name, depot.Name));
796  if (all)
797  isInheritable = (p.Inheritable == true && p.Type == PermType.user && p.Rights == PermRights.all &&
798  String.Equals(p.AppliesTo, user.Principal.Name) && String.Equals(p.Name, depot.Name));
799  return all;
800  }))
801  {
802  string msg = String.Format("{0}{1}", depot.Name, isInheritable ? "+" : String.Empty);
803  canView.Add(msg);
804  }
805  else if (_permissions.Any(
806  delegate(AcPermission p)
807  {
808  return (p.Type == PermType.user && p.Rights == PermRights.none &&
809  String.Equals(p.AppliesTo, user.Principal.Name) && String.Equals(p.Name, depot.Name));
810  }))
811  {
812  ; // user does not have permission to this depot
813  }
814  else // check permissions based on user's group memberships
815  {
816  bool all = false;
817  bool none = _permissions.Any(
818  delegate(AcPermission p)
819  {
820  return (String.Equals(p.Name, depot.Name) && p.Rights == PermRights.none &&
821  p.Type == PermType.group && members.Any(
822  delegate(string m) { return String.Equals(p.AppliesTo, m); }));
823  });
824 
825  // by default everyone has access unless there's a "none" with no "all" override
826  if (none)
827  {
828  all = _permissions.Any(
829  delegate(AcPermission p)
830  {
831  return (String.Equals(p.Name, depot.Name) && p.Rights == PermRights.all &&
832  p.Type == PermType.group && members.Any(
833  delegate(string m) { return String.Equals(p.AppliesTo, m); }));
834  });
835  }
836 
837  if (none && !all)
838  ; // user does not have permission to this depot
839  else // no permission(s) set or an 'all' that overrides one or more 'none' permissions
840  {
841  isInheritable = _permissions.Any(
842  delegate(AcPermission p)
843  {
844  return (p.Inheritable == true && String.Equals(p.Name, depot.Name) &&
845  p.Type == PermType.group && members.Any(
846  delegate(string m) { return String.Equals(p.AppliesTo, m); }));
847  });
848 
849  string msg = String.Format("{0}{1}", depot.Name, isInheritable ? "+" : String.Empty);
850  canView.Add(msg);
851  }
852  }
853  }
854 
855  IEnumerable<string> e = canView.OrderBy(n => n);
856  string depots = String.Join(", ", e);
857  return depots;
858  }
